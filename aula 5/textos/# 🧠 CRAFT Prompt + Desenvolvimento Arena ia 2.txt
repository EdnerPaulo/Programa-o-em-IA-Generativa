## 📋 O PROMPT CRAFT

```
Context:  Um desenvolvedor iniciante quer criar um produto SaaS onde o 
          usuário cola um trecho de código e uma IA explica linha por 
          linha em português simples, traduzindo conceitos técnicos 
          para linguagem acessível.

Role:     Arquiteto de soluções de IA + Engenheiro Full Stack + 
          Product Manager especializado em produtos para iniciantes.

Action:   Desenvolva um plano técnico e de produto completo cobrindo:
          problema, modelagem com TensorFlow/LLMs, deploy, análise 
          competitiva, engenharia de prompt, interface com Streamlit, 
          e pipeline de hospedagem com Git + Render.

Format:   Guia estruturado em 7 etapas com código, diagramas ASCII, 
          tabelas comparativas e exemplos práticos.

Target:   Desenvolvedor solo ou equipe pequena (1-3 pessoas) com 
          conhecimento intermediário em Python, sem experiência 
          prévia em MLOps ou deploy de modelos de linguagem.
```

### Modelo TensorFlow para Fase 2

```python
# arquitetura_treinamento.py
# Fine-tuning de um modelo T5 para explicação de código

import tensorflow as tf
from transformers import (
    TFAutoModelForSeq2SeqLM,
    AutoTokenizer
)

class CodeExplainerModel:
    """
    Modelo seq2seq que recebe código e gera explicação em PT-BR.
    Base: mT5-small (multilíngue, já entende português)
    """

    def __init__(self, model_name="google/mt5-small"):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = TFAutoModelForSeq2SeqLM.from_pretrained(
            model_name
        )

    def preparar_dataset(self, pares_codigo_explicacao):
        """
        pares_codigo_explicacao = [
            {
                "codigo": "x = [i**2 for i in range(10)]",
                "explicacao": "Cria uma lista com os quadrados 
                               dos números de 0 a 9 usando 
                               list comprehension."
            },
            ...
        ]
        """
        entradas = self.tokenizer(
            [p["codigo"] for p in pares_codigo_explicacao],
            max_length=512,
            truncation=True,
            padding="max_length",
            return_tensors="tf"
        )

        saidas = self.tokenizer(
            [p["explicacao"] for p in pares_codigo_explicacao],
            max_length=512,
            truncation=True,
            padding="max_length",
            return_tensors="tf"
        )

        return entradas, saidas

    def treinar(self, entradas, saidas, epochs=3):
        self.model.compile(
            optimizer=tf.keras.optimizers.Adam(
                learning_rate=3e-5
            ),
            loss=tf.keras.losses.SparseCategoricalCrossentropy(
                from_logits=True
            )
        )

        self.model.fit(
            x=dict(entradas),
            y=saidas["input_ids"],
            epochs=epochs,
            batch_size=8
        )

    def explicar(self, codigo: str) -> str:
        inputs = self.tokenizer(
            codigo,
            return_tensors="tf",
            max_length=512,
            truncation=True
        )
        outputs = self.model.generate(
            **inputs,
            max_new_tokens=512,
            temperature=0.3,
            do_sample=True
        )
        return self.tokenizer.decode(
            outputs[0],
            skip_special_tokens=True
        )
```

SYSTEM_PROMPT = """
Você é o "CodeBuddy", um mentor de programação brasileiro

   (ex: "variável é tipo uma caixa com etiqueta")
4. NUNCA use jargão sem explicar antes
5. No final, dê um resumo de 2 frases do que o código faz

7. Sugira um exercício prático relacionado

FORMATO DE SAÍDA:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
📝 CÓDIGO ANALISADO:
[repita o código com numeração]

🔍 EXPLICAÇÃO LINHA A LINHA:
Linha 1 → [explicação]
Linha 2 → [explicação]
...

📦 RESUMO:
[2 frases simples]

🏋️ DESAFIO PRA VOCÊ:
[exercício prático]
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

NÍVEL DO USUÁRIO: {nivel}
"""

NIVEIS = {
    "iniciante": "Nunca programou. Explique como se fosse "
                 "para uma criança de 12 anos.",
    "basico": "Já viu variáveis e if/else. "
              "Pode usar termos simples.",
    "intermediario": "Já fez projetos pequenos. "
                     "Pode ser mais técnico."
}


def build_prompt(codigo: str, nivel: str = "iniciante") -> list:
    return [
        {
            "role": "system",
            "content": SYSTEM_PROMPT.format(
                nivel=NIVEIS.get(nivel, NIVEIS["iniciante"])
            )
        },
        {
            "role": "user",
            "content": f"Explique este código:\n\n```python\n"
                       f"{codigo}\n```"
        }
    ]
```

### Prompt para Parser Linha a Linha (Pré-processamento)

```python
# core/code_parser.py

import ast
import tokenize
import io

class CodeParser:
    """
    Quebra o código em linhas e classifica cada uma.
    Isso enriquece o prompt com metadados.
    """

    def parse(self, codigo: str) -> list[dict]:
        linhas = codigo.strip().split('\n')
        resultado = []

        for i, linha in enumerate(linhas, 1):
            resultado.append({
                "numero": i,
                "conteudo": linha,
                "tipo": self._classificar_linha(linha),
                "indentacao": len(linha) - len(linha.lstrip())
            })

        return resultado

    def _classificar_linha(self, linha: str) -> str:
        linha = linha.strip()

        if not linha:
            return "vazia"
        if linha.startswith('#'):
            return "comentario"
        if linha.startswith('def '):
            return "definicao_funcao"
        if linha.startswith('class '):
            return "definicao_classe"
        if linha.startswith('import ') or \
           linha.startswith('from '):
            return "importacao"
        if linha.startswith('if ') or \
           linha.startswith('elif ') or \
           linha.startswith('else'):
            return "condicional"
        if linha.startswith('for ') or \
           linha.startswith('while '):
            return "loop"
        if linha.startswith('return '):
            return "retorno"
        if linha.startswith('try') or \
           linha.startswith('except'):
            return "tratamento_erro"
        if '=' in linha and not '==' in linha:
            return "atribuicao"
        if linha.startswith('print('):
            return "saida"

        return "expressao"

    def gerar_contexto_rico(self, codigo: str) -> str:
        """
        Gera um contexto estruturado para o LLM
        """
        linhas = self.parse(codigo)
        contexto = "ESTRUTURA DO CÓDIGO:\n"

        for l in linhas:
            if l["tipo"] != "vazia":
                contexto += (
                    f"  Linha {l['numero']} "
                    f"[{l['tipo']}]: {l['conteudo']}\n"
                )

        return contexto
```

                        f"`L{linha['numero']:02d}` "
                        f"{cor} **{linha['tipo']}** → "
                        f"`{linha['conteudo']}`"
                    )

        with tab3:
            total = len(linhas_parseadas)
            tipos = {}
            for l in linhas_parseadas:
                tipos[l["tipo"]] = tipos.get(l["tipo"], 0) + 1

            col_a, col_b, col_c = st.columns(3)
            col_a.metric("Total de linhas", total)
            col_b.metric("Linhas de código", 
                         total - tipos.get("vazia", 0))
            col_c.metric("Complexidade", 
                         "Baixa" if total < 15 else 
                         "Média" if total < 30 else "Alta")

            st.bar_chart(tipos)
```

### Módulo Explainer (core/explainer.py)

```python
# core/explainer.py
import os
from openai import OpenAI

class CodeExplainer:
    """
    Wrapper unificado para múltiplos providers.
    DeepSeek usa API compatível com OpenAI.
    """

    PROVIDERS = {
        "deepseek": {
            "base_url": "https://api.deepseek.com",
            "model": "deepseek-chat",
            "api_key_env": "DEEPSEEK_API_KEY"
        },
        "gemini": {
            "base_url": "https://generativelanguage."
                        "googleapis.com/v1beta/openai/",
            "model": "gemini-2.0-flash",
            "api_key_env": "GEMINI_API_KEY"
        }
    }

    def __init__(self, provider="deepseek"):
        config = self.PROVIDERS[provider]
        self.client = OpenAI(
            base_url=config["base_url"],
            api_key=os.getenv(config["api_key_env"])
        )
        self.model = config["model"]

    def explicar(self, mensagens: list, 
                 contexto_extra: str = "") -> str:

        if contexto_extra:
            mensagens[1]["content"] += (
                f"\n\n{contexto_extra}"
            )

        response = self.client.chat.completions.create(
            model=self.model,
            messages=mensagens,
            temperature=0.3,
            max_tokens=2000
        )

        return response.choices[0].message.content
```
